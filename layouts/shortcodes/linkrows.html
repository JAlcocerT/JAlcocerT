{{- /* Render a modern one-link-per-row list from repeated URL/label pairs. Usage: {{< linkrows "https://example.com/" "Example" "/about/" "About" >}} Homepage example: {{< linkrows "https://www.jalcocertech.com/" "JAlcocerTech ↗" "https://consulting.jalcocertech.com" "Collaborations ↗" "https://jalcocert.github.io/RPi/" "Raspberry Pi Guide ↗" >}} */ -}} {{- $params := .Params -}} {{- $count := len $params -}} {{- if eq $count 0 -}} {{- errorf "linkrows shortcode: expected at least one URL/label pair (page: %s)" .Page.RelPermalink -}} {{- end -}} {{- if ne (mod $count 2) 0 -}} {{- errorf "linkrows shortcode: expected URL/label pairs, got an odd number of arguments on page %s" .Page.RelPermalink -}} {{- end -}}
{{- range $index, $value := $params -}} {{- if eq (mod $index 2) 0 -}} {{- $link := printf "%s" $value -}} {{- $title := printf "%s" (index $params (add $index 1)) -}} {{- $external := strings.HasPrefix $link "http" -}} {{- $href := cond (strings.HasPrefix $link "/") ($link | relURL) $link -}} {{ partial "utils/icon.html" (dict "name" "link") }} {{ $title }} {{ partial "utils/icon.html" (dict "name" "arrow-right") }} {{- end -}} {{- end -}}
