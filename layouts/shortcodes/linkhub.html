{{- /* Render a compact Hextra-style link hub from repeated URL/label pairs. Usage: {{< linkhub "https://example.com/" "Example" "/about/" "About" >}} Homepage example: {{< linkhub "https://www.jalcocertech.com/" "JAlcocerTech ↗" "https://consulting.jalcocertech.com" "Collaborations ↗" "https://jalcocert.github.io/RPi/" "Raspberry Pi Guide ↗" >}} */ -}} {{- $params := .Params -}} {{- $count := len $params -}} {{- if eq $count 0 -}} {{- errorf "linkhub shortcode: expected at least one URL/label pair (page: %s)" .Page.RelPermalink -}} {{- end -}} {{- if ne (mod $count 2) 0 -}} {{- errorf "linkhub shortcode: expected URL/label pairs, got an odd number of arguments on page %s" .Page.RelPermalink -}} {{- end -}}
